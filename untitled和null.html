null vs undefined 详解

核心区别总结

  • null: 表示"故意设置为空值"，是一个明确的没有对象值的值
  • undefined: 表示"未定义"，是变量声明了但未赋值时的默认值

1. 基本定义

// null - 明确表示"无值"或"空值" let emptyValue = null; // undefined - 表示变量已声明但未赋值 let uninitializedVar; // undefined

2. 类型检查

console.log(typeof null); // "object" (这是JavaScript的一个历史bug) console.log(typeof undefined); // "undefined" console.log(null instanceof Object); // false

3. 赋值场景

// undefined 出现的场景： let a; // 声明但未初始化 let b = {x: 1}; console.log(b.y); // 访问不存在的对象属性 function f() {} console.log(f()); // 函数没有返回值 function g(x) { console.log(x); } g(); // 函数参数未传递 // null 出现的场景： let c = null; // 明确设置为无值

4. 比较运算

// == 比较 (相等) console.log(null == undefined); // true console.log(null == 0); // true console.log(undefined == 0); // false // === 比较 (严格相等) console.log(null === undefined); // false console.log(null === null); // true console.log(undefined === undefined); // true

5. 实际应用场景

// null 的典型用途： function findUser(id) { // 如果找不到用户，返回 null if (!validId(id)) { return null; } return getUserById(id); } // undefined 的典型用途： function processOptions(options) { // 检查选项是否被提供 if (options === undefined) { options = getDefaultOptions(); } }

6. 检测方法

// 检测 undefined let x; if (typeof x === 'undefined') { /* x 未定义 */ } if (x === undefined) { /* x 未定义 */ } // 检测 null if (x === null) { /* x 为 null */ } // 同时检测 null 或 undefined (falsy 值) if (!x) { /* x 为 falsy 值，包括 null, undefined, 0, "", false, NaN */ } if (x == null) { /* x 为 null 或 undefined */ }

7. 最佳实践

  • 如果想表示"无值"，显式地使用 null
  • 不要显式地将变量设为 undefined，那是语言引擎的工作
  • 使用 === 进行严格比较，避免类型转换带来的问题
  • 当需要检查值是否存在时，同时检查 null 和 undefined
  • 使用 ?? (空值合并运算符) 来处理 null 和 undefined

8. 空值合并运算符示例

let user = null; let defaultName = "匿名用户"; // 空值合并运算符：仅当左侧为 null 或 undefined 时返回右侧值 let displayName = user ?? defaultName; // "匿名用户" let zeroUser = 0; let displayZero = zeroUser ?? "默认值"; // 0 (因为 0 不是 null 或 undefined)
